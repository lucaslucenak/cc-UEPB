Cliente ( _mat, nome, profissao, end, sexo, email, telefone)
CREATE TABLE Cliente (
	mat SERIAL,
	nome varchar(255),
	profissao varchar(100),
	endereco varchar(255),
	sexo varchar(30),
	email varchar(100),
	telefone varchar(20),
	PRIMARY KEY (mat)
	);

Dependente (_id, #matCliente, nome, endereco, sexo, email)
CREATE TABLE Dependente (
	id SERIAL,
	matCliente int,
	nome varchar(255),
	endereco varchar(255),
	sexo varchar(30),	
	email varchar(100),
	PRIMARY KEY (id),
	FOREIGN KEY (matCliente) REFERENCES cliente(mat)
	);
